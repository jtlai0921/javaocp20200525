class student
{
	String name;
	int chi;
	int eng;
	int sum;

	student(String name,int chi,int eng)
	{
		this.name=name;
		this.chi=chi;
		this.eng=eng;

		sum=chi+eng;
	}


	void change1(int chi,int eng)
	{
		this.chi=chi;
		this.eng=eng;
		sum=chi+eng;
	}

	int change2(int chi,int eng)
	{
		this.chi=chi;
		this.eng=eng;
		sum=chi+eng;

		return sum;
	}


	void show()
	{
		System.out.println("名:"+name+
			"\t國文:"+chi+
			"\t英文:"+eng+
			"\t總分:"+sum);
	}
}
