package exceptionEX;

class check extends Exception
{
	check(String msg)
	{
		System.out.println(msg);
	}
}



public class school {
	private String name;
	private int chi;
	private int eng;
	public school(String name, int chi, int eng) throws check{
		
		if(chi>=0 && chi<=100 && eng>=0 && eng<=100)
		{
			this.name = name;
			this.chi = chi;
			this.eng = eng;
		}
		else
		{
			
				throw new check("新增的分數需介於0~100");
			
		}
	}
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public int getChi() {
		return chi;
	}
	public void setChi(int chi) throws check {
		if(chi>=0 && chi<=100)
		{
		this.chi = chi;
		}
		else
		{
			throw new check("修改國文的分數需介於0~100");
		}
		
		
	}
	public int getEng() {
		return eng;
	}
	public void setEng(int eng) {
		this.eng = eng;
	}
	
	
	

}
